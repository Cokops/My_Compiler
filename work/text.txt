int main() {
    int a = 50;
    int b = 30;
    int c = 40;
    int max;
    
    if (a > b) {
        if (a > c) {
            max = a;
        } else {
            max = c;
        }
    } else {
        if (b > c) {
            max = b;
        } else {
            max = c;
        }
    }
    return max;
}
// Ожидаемый результат: 50
